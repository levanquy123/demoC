#include <stdio.h>

int main(int argc, char const *argv[])
{
    int arr1[10];
    float arr2[5];

    for (int i = 0; i < 5; i++)
    {
        printf("Nhap arr[%d]: ", i);
        scanf("%f", &arr2[i]);
    }
    int index=0;
    for (int j = 0; j < 10;j++)
    {
        if (j % 2 != 0)
        {
            arr1[j] = arr2[index];
            index++;
           
        }
        else
        {
            arr1[j] = 0;
        }
    }
    for (int k = 0; k < 5; k++)
    {
        printf("FloatArr[%d]= %.f\n", k, arr2[k]);
    }
    for (int l = 0; l < 10; l++)
    {
        printf("IntArr[%d]= %d\n", l, arr1[l]);
    }

    return 0;
}
