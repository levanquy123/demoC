#include <stdio.h>
#include<math.h>

int nt(int m)
{
    for(int i=2;i<=sqrt(m);i++)
    if(m%i==0) return 0;
    return 1;
}
int main(){
     int i = 2,n,dem = 0;
     printf("\n Nhap n: ");
     scanf("%d",&n);
     while(dem<n){
         if(nt(i)){

             dem++;
         }

     }
 }
