#include <stdio.h>

int main() 
{
    int day;
    int month;
    int year;
    printf("Input day: ");
    scanf("%d", &day);
    printf("Input month: ");
    scanf("%d", &month);
    printf("Input year: ");
    scanf("%d", &year);
    printf("Today is: %d/%d/%d\n", day, month, year);
    return 0;
}
