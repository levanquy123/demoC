#include <stdio.h>

int main() 
{  
    int a;
    int b;
    int add;
    int sub;
    int mul;
    int div;
    printf("Input a: ");
    scanf("%d", &a);
    printf("Input b: ");
    scanf("%d", &b);
    add = a + b;
    sub = a - b;
    mul = a * b;
    div = a / b;
    printf("a + b = %d", add);
    printf("\na - b = %d", sub);
    printf("\na * b = %d", mul);
    printf("\na / b = %d", div);
    printf("\n");
    return 0;
}
